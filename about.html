'1' : '0'; ctx.fillText(bin, i * 18, yPos[i] * 18 + 18); if (Math.random() > 0.975) { yPos[i] = 0; } else { yPos[i] = (yPos[i] + 1) % Math.floor(h / 18); } } requestAnimationFrame(draw); } draw(); }
